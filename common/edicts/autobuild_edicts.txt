planet_edict = {
	name = hide_planetary_edicts
	influence_cost = 0
	length = 0
	potential = {}

	effect = {
		if = {
			limit = {
				has_global_flag  = hide_autobuild_planetary_edicts
			}
			remove_global_flag  = hide_autobuild_planetary_edicts
			else = {
				set_global_flag = hide_autobuild_planetary_edicts
			}
		}
	}
	ai_weight = {
		weight = 0
	}
}
